140  std::vector<Texture> Model::loadMaterialTextures(aiMaterial* mat, aiTextureType type, std::string typeName)
141  {
142  std::vector<Texture> textures;
143  for (unsigned int i = 0; i < mat->GetTextureCount(type); i++)
144  {
145  aiString str;
146  mat->GetTexture(type, i, &str);
147  // check if texture was loaded before and if so, continue to next iteration: skip loading a new texture
148  bool skip = false;
149  for (unsigned int j = 0; j < textures_loaded.size(); j++)
150  {
151  if (std::strcmp(textures_loaded[j].path.data(), str.C_Str()) == 0)
152  {
153  textures.push_back(textures_loaded[j]);
154  skip = true; // a texture with the same filepath has already been loaded, continue to next one. (optimization)
155  break;
156  }
157  }
158  if (!skip)
159  { // if texture hasn't been loaded already, load it
160  Texture texture;

162  texture.type = typeName;
163  texture.path = str.C_Str();
164  textures.push_back(texture);
165  textures_loaded.push_back(texture); // store it as texture loaded for entire model, to ensure we won't unnecesery load duplicate textures.
166  }
